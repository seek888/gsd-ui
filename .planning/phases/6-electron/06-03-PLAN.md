### Task 1: 创建 Store 初始化模块
创建 `src/stores/electronStore.ts`:
```typescript
import { electronAPI } from '@/lib/electronAPI'

const isElectron = typeof window !== 'undefined' && window.electronAPI !== undefined

export async function getStoredValue<T>(key: string, defaultValue: T): Promise<T> {
  if (!isElectron) return defaultValue
  try {
    const value = await electronAPI.getSetting<T>(key)
    return value ?? defaultValue
  } catch {
    return defaultValue
  }
}

export async function setStoredValue<T>(key: string, value: T): Promise<void> {
  if (!isElectron) return
  try {
    await electronAPI.setSetting(key, value)
  } catch (err) {
    console.error('Failed to save setting:', key, err)
  }
}
```

### Task 2: 创建 Electron API 包装器
创建 `src/lib/electronAPI.ts`:
```typescript
// 运行时检查，避免 SSR 问题
let api: typeof window.electronAPI | null = null

if (typeof window !== 'undefined' && window.electronAPI) {
  api = window.electronAPI
}

export const electronAPI = {
  // Shell
  async executeCommand(program: string, args: string[], cwd?: string) {
    if (!api) throw new Error('Electron API not available')
    return api.executeCommand(program, args, cwd)
  },
  
  async killCommand(pid: number) {
    if (!api) throw new Error('Electron API not available')
    return api.killCommand(pid)
  },
  
  onCommandOutput(callback: (data: { type: string; data: string }) => void) {
    if (!api) return () => {}
    return api.onCommandOutput(callback)
  },
  
  // File System
  async readFile(path: string) {
    if (!api) throw new Error('Electron API not available')
    return api.readFile(path)
  },
  
  async writeFile(path: string, content: string) {
    if (!api) throw new Error('Electron API not available')
    return api.writeFile(path, content)
  },
  
  async readDir(path: string) {
    if (!api) throw new Error('Electron API not available')
    return api.readDir(path)
  },
  
  async exists(path: string) {
    if (!api) throw new Error('Electron API not available')
    return api.exists(path)
  },
  
  // Dialogs
  async openDirectory() {
    if (!api) throw new Error('Electron API not available')
    return api.openDirectory()
  },
  
  async showSaveDialog(options: any) {
    if (!api) throw new Error('Electron API not available')
    return api.showSaveDialog(options)
  },
  
  async showMessageBox(options: any) {
    if (!api) throw new Error('Electron API not available')
    return api.showMessageBox(options)
  },
  
  // Settings
  async getSetting<T>(key: string): Promise<T | null> {
    if (!api) return null
    return api.getSetting<T>(key)
  },
  
  async setSetting<T>(key: string, value: T) {
    if (!api) return
    return api.setSetting(key, value)
  },
  
  get platform() {
    return api?.platform ?? 'unknown'
  }
}
```

### Task 3: 迁移 projectStore
更新 `src/stores/projectStore.ts`:
```typescript
import { create } from 'zustand'
import { electronAPI } from '@/lib/electronAPI'
import { getStoredValue, setStoredValue } from './electronStore'

interface ProjectState {
  projectPath: string | null
  isLoading: boolean
  error: string | null
}

interface ProjectActions {
  setProjectPath: (path: string | null) => Promise<void>
  selectDirectory: () => Promise<void>
  loadProjectPath: () => Promise<void>
}

export const useProjectStore = create<ProjectState & ProjectActions>((set, get) => ({
  projectPath: null,
  isLoading: false,
  error: null,
  
  loadProjectPath: async () => {
    const path = await getStoredValue<string | null>('projectPath', null)
    set({ projectPath: path })
  },
  
  setProjectPath: async (path) => {
    set({ projectPath: path })
    await setStoredValue('projectPath', path)
  },
  
  selectDirectory: async () => {
    try {
      set({ isLoading: true, error: null })
      const path = await electronAPI.openDirectory()
      if (path) {
        await get().setProjectPath(path)
      }
    } catch (err) {
      set({ error: err instanceof Error ? err.message : 'Failed to select directory' })
    } finally {
      set({ isLoading: false })
    }
  }
}))
```

### Task 4: 迁移 uiStore
更新 `src/stores/uiStore.ts`:
```typescript
import { create } from 'zustand'
import { getStoredValue, setStoredValue } from './electronStore'

interface UIState {
  sidebarCollapsed: boolean
  currentView: 'dashboard' | 'documents' | 'terminal' | 'settings'
  terminalHeight: number
}

interface UIActions {
  toggleSidebar: () => Promise<void>
  setCurrentView: (view: UIState['currentView']) => Promise<void>
  setTerminalHeight: (height: number) => Promise<void>
  loadUIState: () => Promise<void>
}

export const useUIStore = create<UIState & UIActions>((set) => ({
  sidebarCollapsed: false,
  currentView: 'dashboard',
  terminalHeight: 300,
  
  loadUIState: async () => {
    const sidebarCollapsed = await getStoredValue('sidebarCollapsed', false)
    const currentView = await getStoredValue<'dashboard' | 'documents' | 'terminal' | 'settings'>('currentView', 'dashboard')
    const terminalHeight = await getStoredValue('terminalHeight', 300)
    set({ sidebarCollapsed, currentView, terminalHeight })
  },
  
  toggleSidebar: async () => {
    const newValue = !get().sidebarCollapsed
    set({ sidebarCollapsed: newValue })
    await setStoredValue('sidebarCollapsed', newValue)
  },
  
  setCurrentView: async (view) => {
    set({ currentView: view })
    await setStoredValue('currentView', view)
  },
  
  setTerminalHeight: async (height) => {
    set({ terminalHeight: height })
    await setStoredValue('terminalHeight', height)
  }
}))
```
