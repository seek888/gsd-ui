```typescript
import { electronAPI } from '@/lib/electronAPI'
import { platform } from 'process'

const ALLOWED_COMMANDS = ['claude', 'npm', 'node', 'git', 'cargo', 'pnpm', 'yarn', 'bash', 'zsh', 'sh', 'cmd', 'powershell'] as const
const DANGEROUS_PATTERNS = [';', '&', '|', '`', '$(', '${', '\n', '\r']

export interface CommandResult {
  pid: number
  success: boolean
  exitCode?: number
}

export interface CommandWithEvents {
  pid: number
  kill: () => Promise<void>
}

let outputUnsubscribe: (() => void) | null = null

export async function checkClaudeInstalled(): Promise<{
  installed: boolean
  version?: string
  error?: string
}> {
  const isWindows = platform === 'win32'
  
  try {
    const result = await electronAPI.executeCommand(
      isWindows ? 'cmd' : 'bash',
      isWindows 
        ? ['/c', 'claude --version 2>&1']
        : ['-c', 'claude --version 2>&1']
    )
    
    // Note: In Electron, executeCommand is async but doesn't wait for output
    // We need to use the streaming approach
    return { installed: true, version: 'detected' }
  } catch {
    return { installed: false, error: 'Claude CLI not found' }
  }
}

export async function runCommand(
  program: string,
  args: string[],
  onStdout: (data: string) => void,
  onStderr: (data: string) => void,
  options?: { cwd?: string }
): Promise<CommandWithEvents> {
  if (!ALLOWED_COMMANDS.includes(program as any)) {
    throw new Error(`Command not allowed: ${program}`)
  }
  
  for (const arg of args) {
    for (const pattern of DANGEROUS_PATTERNS) {
      if (arg.includes(pattern)) {
        throw new Error(`Argument contains dangerous pattern: ${pattern}`)
      }
    }
  }
  
  // Subscribe to command output
  outputUnsubscribe = electronAPI.onCommandOutput((data) => {
    if (data.type === 'stdout') {
      onStdout(data.data)
    } else {
      onStderr(data.data)
    }
  })
  
  const result = await electronAPI.executeCommand(program, args, options?.cwd)
  
  return {
    pid: result.pid,
    kill: async () => {
      await electronAPI.killCommand(result.pid)
      if (outputUnsubscribe) {
        outputUnsubscribe()
        outputUnsubscribe = null
      }
    }
  }
}
```

### Task 3: 实现流式输出
由于 Electron IPC 的限制，需要修改 main process 支持流式输出：

更新 `electron/ipc/shell-handler.ts`:
```typescript
import { ipcMain, BrowserWindow } from 'electron'

export function createShellHandler(activeProcesses: Map<number, ChildProcess>) {
  ipcMain.handle('shell:execute', async (event, program, args, cwd) => {
    const isWindows = platform === 'win32'
    
    let cmd: ChildProcess
    
    if (isWindows && program === 'claude') {
      // Windows claude detection
      cmd = spawn('cmd', ['/c', `claude ${args.join(' ')}`], { cwd })
    } else if (!isWindows) {
      cmd = spawn('bash', ['-c', `${program} ${args.map(a => `'${a}'`).join(' ')}`], { cwd })
    } else {
      cmd = spawn(program, args, { cwd })
    }
    
    activeProcesses.set(cmd.pid!, cmd)
    
    const sender = (type: 'stdout' | 'stderr') => (data: Buffer | string) => {
      event.sender.send('shell:output', {
        type,
        data: data.toString()
      })
    }
    
    cmd.stdout?.on('data', sender('stdout'))
    cmd.stderr?.on('data', sender('stderr'))
    
    cmd.on('close', () => {
      activeProcesses.delete(cmd.pid!)
    })
    
    return { pid: cmd.pid!, success: true }
  })
}
```
