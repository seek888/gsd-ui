  <action>
    Create src/lib/watchers.ts with debounced file watching utilities.

    Import watch from @tauri-apps/plugin-fs.

    Export functions:

    1. createDebouncedCallback(fn: () => void, delay: number): (() => void, () => void)
       - Returns tuple: [debouncedFn, cancel]
       - debouncedFn: clears previous timeout, sets new timeout for delay ms
       - cancel: clears pending timeout
       - Implementation: standard setTimeout/clearTimeout pattern

    2. watchDirectory(path: string, onChange: (event: { type: string; path: string }) => void, debounceMs: number): Promise<() => void>
       - Calls fs.watch(path, { recursive: true }, onChange)
       - Wraps onChange in createDebouncedCallback
       - Returns cleanup function (unwatch from fs.watch)
       - Per D-10: 500ms debounce

    Implementation:
    ```typescript
    import { watch } from '@tauri-apps/plugin-fs';

    export function createDebouncedCallback<T extends (...args: any[]) => void>(
      fn: T,
      delay: number
    ): [(...args: Parameters<T>) => void, () => void] {
      let timeoutId: ReturnType<typeof setTimeout> | null = null;

      const debouncedFn = (...args: Parameters<T>) => {
        if (timeoutId) clearTimeout(timeoutId);
        timeoutId = setTimeout(() => fn(...args), delay);
      };

      const cancel = () => {
        if (timeoutId) clearTimeout(timeoutId);
        timeoutId = null;
      };

      return [debouncedFn, cancel];
    }

    export async function watchDirectory(
      path: string,
      onChange: (event: { type: string; path: string }) => void,
      debounceMs: number = 500
    ): Promise<() => void> {
      const [debouncedOnChange] = createDebouncedCallback(onChange, debounceMs);
      const unwatch = await watch(path, { recursive: true }, (event) => {
        debouncedOnChange({ type: event.type, path: event.path });
      });
      return unwatch;
    }
    ```

    Per RESEARCH.md: Tauri fs.watch() includes built-in debouncing via notify-debouncer-full, but we add JavaScript-side debounce for consistent 500ms timing.
