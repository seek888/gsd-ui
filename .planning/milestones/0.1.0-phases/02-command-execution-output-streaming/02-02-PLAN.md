```typescript
commandExitStatus: new Map(),

trackCommandExit: (commandId, success, exitCode) => set((state) => {
  const newMap = new Map(state.commandExitStatus);
  newMap.set(commandId, { success, exitCode, timestamp: Date.now() });
  return { commandExitStatus: newMap };
}),

getCommandStatus: (commandId) => get().commandExitStatus?.get(commandId),

clearCommandStatus: (commandId) => set((state) => {
  const newMap = new Map(state.commandExitStatus);
  newMap.delete(commandId);
  return { commandExitStatus: newMap };
}),
```
