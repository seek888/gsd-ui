    ```typescript
    export interface Phase {

      name: string;         // 例如 "Progress & State Views"
      status: PlanStatus;   // 阶段状态
    ```

    ```typescript
    import { readTextFile } from '@tauri-apps/plugin-fs';
    import { runCommand } from '@/lib/shell';
    import type {
      Phase, Plan, SessionState, RoadmapData,
      GsdToolsRoadmapOutput, GsdToolsStateOutput
    } from '@/types/progress';

    /**
     * 执行 gsd-tools.cjs 命令并解析 JSON 输出
     * 返回 Promise<{ success: boolean; data?: T; error?: string }>
     */
    async function executeGsdTool<T>(
      projectPath: string,
      command: 'roadmap analyze' | 'state-snapshot'
    ): Promise<{ success: boolean; data?: T; error?: string }> {
      const gsdToolsPath = `${projectPath}/.planning/bin/gsd-tools.cjs`;

      try {
        // 收集 stdout 输出
        let stdoutOutput = '';
        let stderrOutput = '';

        const { child, command: cmd, onClose } = await runCommand(
          'node',
          [gsdToolsPath, command],
          (data) => { stdoutOutput += data; },
          (data) => { stderrOutput += data; }
        );

        // 等待命令完成
        return new Promise((resolve) => {
          const timeout = setTimeout(() => {
            child.kill();
            resolve({ success: false, error: 'Command timeout after 10s' });
          }, 10000);

          onClose(({ code }) => {
            clearTimeout(timeout);
            if (code === 0) {
              try {
                const json = JSON.parse(stdoutOutput.trim());
                resolve({ success: true, data: json as T });
              } catch (parseErr) {
                resolve({
                  success: false,
                  error: `Failed to parse JSON output: ${parseErr instanceof Error ? parseErr.message : String(parseErr)}`
                });
              }
            } else {
              resolve({
                success: false,
                error: stderrOutput.trim() || `Command exited with code ${code}`
              });
            }
          });
        });
      } catch (err) {
        return {
          success: false,
          error: err instanceof Error ? err.message : String(err)
        };
      }
    }

    /**
     * 转换 gsd-tools 输出为内部 RoadmapData 格式
     */
    function convertRoadmapOutput(output: GsdToolsRoadmapOutput): RoadmapData {
      const phases: Phase[] = output.phases.map((p) => {
        const plans: Plan[] = p.plans.map((pl) => ({
          id: pl.id,
          name: pl.name,
          status: pl.hasSummaryMd ? 'complete' : (pl.hasPlanMd ? 'in-progress' : 'not-started'),
          hasPlanMd: pl.hasPlanMd,
          hasSummaryMd: pl.hasSummaryMd,
        }));

        const completedPlans = plans.filter(pl => pl.status === 'complete').length;

        return {
          number: p.number,
          slug: p.slug,
          name: p.name,
          status: completedPlans === plans.length ? 'complete' :
                  (completedPlans > 0 ? 'in-progress' : 'not-started'),
          progress: plans.length > 0 ? Math.round((completedPlans / plans.length) * 100) : 0,
          plansCount: plans.length,
          completedPlans,
          plans,
          isActive: p.number === output.currentPhaseNumber,
        };
      });

      const totalPlans = phases.reduce((sum, p) => sum + p.plansCount, 0);
      const completedPlans = phases.reduce((sum, p) => sum + p.completedPlans, 0);
      const overallProgress = totalPlans > 0 ? Math.round((completedPlans / totalPlans) * 100) : 0;

      return {
        phases,
        totalPhases: phases.length,
        totalPlans,
        completedPlans,
        overallProgress,
        currentPhaseNumber: output.currentPhaseNumber,
      };
    }

    /**
     * 转换 gsd-tools state 输出为内部 SessionState 格式
     */
    function convertStateOutput(output: GsdToolsStateOutput): SessionState {
      return {
        currentPhase: output.currentPhase,
        currentPlan: output.currentPlan,
        lastActivity: output.lastActivity || '无最近活动',
        blockers: output.blockers?.map((b, i) => ({
          id: `blocker-${i}`,
          description: b.description,
          severity: b.severity || 'medium',
        })) || [],
        decisions: output.decisions?.map((d) => ({
          id: d.id,
          description: d.description,
          phase: d.phase,
        })) || [],
        recentActivity: [], // gsd-tools state-snapshot 可能不返回此字段
      };
    }

    /**
     * 解析 ROADMAP.md 文件，提取阶段和计划数据（备用方案）
     * 仅在 gsd-tools CLI 不可用时使用
     */
    function parseRoadmapMarkdown(content: string): RoadmapData {
      const lines = content.split('\n');
      const phases: Phase[] = [];
      let currentPhase: Partial<Phase> | null = null;
      let currentPlans: Plan[] = [];

      for (let i = 0; i < lines.length; i++) {
        const line = lines[i];

        // 匹配阶段标题：### Phase X: Name
        const phaseMatch = line.match(/^### Phase (\d+): (.+)/);
        if (phaseMatch) {
          // 保存上一个阶段
          if (currentPhase && currentPhase.name) {
            phases.push({
              number: currentPhase.number!,
              slug: currentPhase.slug || generateSlug(currentPhase.name),
              name: currentPhase.name,
              status: determinePhaseStatus(currentPlans),
              progress: calculatePhaseProgress(currentPlans),
              plansCount: currentPlans.length,
              completedPlans: currentPlans.filter(p => p.status === 'complete').length,
              plans: currentPlans,
              isActive: false, // 稍后计算
            });
          }

          const phaseNum = parseInt(phaseMatch[1], 10);
          const phaseName = phaseMatch[2].trim();
          currentPhase = { number: phaseNum, name: phaseName, slug: generateSlug(phaseName) };
          currentPlans = [];
          continue;
        }

        // 匹配计划列表项
        const planMatch = line.match(/^- \[([ x])\] (\d+-\d+): (.+)/);
        if (planMatch && currentPhase) {
          const isComplete = planMatch[1] === 'x';
          const planId = planMatch[2];
          const planName = planMatch[3].trim();

          currentPlans.push({
            id: planId,
            name: planName,
            status: isComplete ? 'complete' : 'not-started',
            hasPlanMd: true,
            hasSummaryMd: isComplete,
          });
        }
      }

      // 添加最后一个阶段
      if (currentPhase && currentPhase.name) {
        phases.push({
          number: currentPhase.number!,
          slug: currentPhase.slug || generateSlug(currentPhase.name),
          name: currentPhase.name,
          status: determinePhaseStatus(currentPlans),
          progress: calculatePhaseProgress(currentPlans),
          plansCount: currentPlans.length,
          completedPlans: currentPlans.filter(p => p.status === 'complete').length,
          plans: currentPlans,
          isActive: false,
        });
      }

      // 计算当前活跃阶段
      const lastIncompletePhase = phases.find(p => p.status !== 'complete');
      const currentPhaseNumber = lastIncompletePhase ? lastIncompletePhase.number : phases.length;

      // 标记活跃阶段
      phases.forEach(p => {
        p.isActive = p.number === currentPhaseNumber;
      });

      // 计算整体进度
      const totalPlans = phases.reduce((sum, p) => sum + p.plansCount, 0);
      const completedPlans = phases.reduce((sum, p) => sum + p.completedPlans, 0);
      const overallProgress = totalPlans > 0 ? Math.round((completedPlans / totalPlans) * 100) : 0;

      return {
        phases,
        totalPhases: phases.length,
        totalPlans,
        completedPlans,
        overallProgress,
        currentPhaseNumber,
      };
    }

    /**
     * 解析 STATE.md 文件，提取会话状态（备用方案）
     */
    async function parseStateMarkdown(projectPath: string): Promise<SessionState> {
      const statePath = `${projectPath}/.planning/STATE.md`;

      try {
        const content = await readTextFile(statePath);
        const lines = content.split('\n');
        const state: SessionState = {
          currentPhase: 1,
          currentPlan: null,
          lastActivity: '',
          blockers: [],
          decisions: [],
          recentActivity: [],
        };

        let inBlockers = false;
        let inDecisions = false;

        for (let i = 0; i < lines.length; i++) {
          const line = lines[i];

          // 解析 Phase: X
          const phaseMatch = line.match(/^Phase:\s*(\d+)/);
          if (phaseMatch) {
            state.currentPhase = parseInt(phaseMatch[1], 10);
            continue;
          }

          // 解析 Plan: Y
          const planMatch = line.match(/^Plan:\s*(.+)/);
          if (planMatch) {
            const planValue = planMatch[1].trim();
            if (planValue !== 'Not started') {
              state.currentPlan = parseInt(planValue.replace(/\D/g, ''), 10) || null;
            }
            continue;
          }

          // 解析 Last activity
          const activityMatch = line.match(/^Last activity:\s*(.+)/);
          if (activityMatch) {
            state.lastActivity = activityMatch[1].trim();
            continue;
          }

          // 检测章节
          if (line.includes('Blockers') || line.includes('Concerns')) {
            inBlockers = true;
            inDecisions = false;
            continue;
          }
          if (line.includes('Decisions')) {
            inDecisions = true;
            inBlockers = false;
            continue;
          }

          // 解析阻塞项
          if (inBlockers && line.trim() && !line.startsWith('#')) {
            if (!line.toLowerCase().includes('none') && !line.toLowerCase().includes('无') && line.startsWith('-')) {
              state.blockers.push({
                id: `blocker-${state.blockers.length}`,
                description: line.replace(/^-\s*/, '').trim(),
                severity: 'medium',
              });
            }
          }

          // 解析决策
          if (inDecisions && line.trim() && !line.startsWith('#') && line.startsWith('-')) {
            state.decisions.push({
              id: `decision-${state.decisions.length}`,
              description: line.replace(/^-\s*/, '').trim(),
            });
          }
        }

        return state;
      } catch (error) {
        console.error('Failed to read STATE.md:', error);
        return {
          currentPhase: 1,
          currentPlan: null,
          lastActivity: '无法加载会话状态',
          blockers: [],
          decisions: [],
          recentActivity: [],
        };
      }
    }

    /**
     * 解析 ROADMAP.md 文件（备用方案的入口）
     */
    async function fallbackAnalyzeRoadmap(projectPath: string): Promise<RoadmapData> {
      const roadmapPath = `${projectPath}/.planning/ROADMAP.md`;
      const content = await readTextFile(roadmapPath);
      return parseRoadmapMarkdown(content);
    }

    /**
     * 获取 Roadmap 数据（优先使用 gsd-tools CLI，失败时回退到 markdown 解析）
     * 对应决策 D-15: Phase progress from `node gsd-tools.cjs roadmap analyze`
     */
    export async function analyzeRoadmap(projectPath: string): Promise<RoadmapData> {
      // 优先尝试 gsd-tools CLI
      const cliResult = await executeGsdTool<GsdToolsRoadmapOutput>(projectPath, 'roadmap analyze');

      if (cliResult.success && cliResult.data) {
        return convertRoadmapOutput(cliResult.data);
      }

      // CLI 失败，回退到 markdown 解析
      console.warn('gsd-tools CLI failed, falling back to markdown parsing:', cliResult.error);
      try {
        return await fallbackAnalyzeRoadmap(projectPath);
      } catch (fallbackError) {
        throw new Error(
          `Failed to load roadmap data. CLI error: ${cliResult.error || 'unknown'}. ` +
          `Fallback error: ${fallbackError instanceof Error ? fallbackError.message : String(fallbackError)}`
        );
      }
    }

    /**
     * 获取 Session State 数据（优先使用 gsd-tools CLI，失败时回退到 markdown 解析）
     * 对应决策 D-16: Session state from `node gsd-tools.cjs state-snapshot`
     */
    export async function getStateSnapshot(projectPath: string): Promise<SessionState> {
      // 优先尝试 gsd-tools CLI
      const cliResult = await executeGsdTool<GsdToolsStateOutput>(projectPath, 'state-snapshot');

      if (cliResult.success && cliResult.data) {
        return convertStateOutput(cliResult.data);
      }

      // CLI 失败，回退到 markdown 解析
      console.warn('gsd-tools CLI failed, falling back to markdown parsing:', cliResult.error);
      try {
        return await parseStateMarkdown(projectPath);
      } catch (fallbackError) {
        // 返回默认状态而不是抛出错误
        return {
          currentPhase: 1,
          currentPlan: null,
          lastActivity: '无法加载会话状态',
          blockers: [],
          decisions: [],
          recentActivity: [],
        };
      }
    }

    /**
     * 根据计划列表确定阶段状态（备用方案）
     */
    function determinePhaseStatus(plans: Plan[]): Phase['status'] {
      if (plans.length === 0) return 'not-started';
      const allComplete = plans.every(p => p.status === 'complete');
      const anyComplete = plans.some(p => p.status === 'complete');
      if (allComplete) return 'complete';
      if (anyComplete) return 'in-progress';
      return 'not-started';
    }

    /**
     * 计算阶段进度百分比（备用方案）
     */
    function calculatePhaseProgress(plans: Plan[]): number {
      if (plans.length === 0) return 0;
      const completed = plans.filter(p => p.status === 'complete').length;
      return Math.round((completed / plans.length) * 100);
    }

    /**
     * 生成 URL 友好的 slug（备用方案）
     */
    function generateSlug(text: string): string {
      return text
        .toLowerCase()
        .replace(/[^a-z0-9]+/g, '-')
        .replace(/^-|-$/g, '');
    }
    ```

    ```typescript
    import { create } from 'zustand';
    import type { Phase, SessionState, RoadmapData, LoadingState } from '@/types/progress';
    import { analyzeRoadmap, getStateSnapshot } from '@/lib/gsd-tools';

    interface ProgressState {
      // 数据状态
      roadmapData: RoadmapData | null;
      sessionState: SessionState | null;
      loadingState: LoadingState;
      error: string | null;

      // 展开状态（用于手风琴）
      expandedPhases: Set<number>;

      // Actions
      loadProgressData: (projectPath: string) => Promise<void>;
      refreshData: (projectPath: string) => Promise<void>;
      togglePhaseExpansion: (phaseNumber: number) => void;
      expandPhase: (phaseNumber: number) => void;
      collapsePhase: (phaseNumber: number) => void;
      expandAll: () => void;
      collapseAll: () => void;
      clearError: () => void;
    }

    export const useProgressStore = create<ProgressState>((set, get) => ({
      // 初始状态
      roadmapData: null,
      sessionState: null,
      loadingState: 'idle',
      error: null,
      expandedPhases: new Set<number>(),

      /**
       * 加载进度数据（首次加载或刷新）
       */
      loadProgressData: async (projectPath: string) => {
        set({ loadingState: 'loading', error: null });

        try {
          // 并行加载 roadmap 和 state 数据
          const [roadmapData, sessionState] = await Promise.all([
            analyzeRoadmap(projectPath),
            getStateSnapshot(projectPath),
          ]);

          set({
            roadmapData,
            sessionState,
            loadingState: 'success',
            error: null,
          });
        } catch (err) {
          const message = err instanceof Error ? err.message : String(err);
          set({
            loadingState: 'error',
            error: `加载进度数据失败: ${message}`,
            roadmapData: null,
            sessionState: null,
          });
        }
      },

      /**
       * 刷新数据（重新加载）
       */
      refreshData: async (projectPath: string) => {
        await get().loadProgressData(projectPath);
      },

      /**
       * 切换阶段展开/收起状态
       */
      togglePhaseExpansion: (phaseNumber: number) => {
        set((state) => {
          const newExpanded = new Set(state.expandedPhases);
          if (newExpanded.has(phaseNumber)) {
            newExpanded.delete(phaseNumber);
          } else {
            newExpanded.add(phaseNumber);
          }
          return { expandedPhases: newExpanded };
        });
      },

      /**
       * 展开指定阶段
       */
      expandPhase: (phaseNumber: number) => {
        set((state) => {
          const newExpanded = new Set(state.expandedPhases);
          newExpanded.add(phaseNumber);
          return { expandedPhases: newExpanded };
        });
      },

      /**
       * 收起指定阶段
       */
      collapsePhase: (phaseNumber: number) => {
        set((state) => {
          const newExpanded = new Set(state.expandedPhases);
          newExpanded.delete(phaseNumber);
          return { expandedPhases: newExpanded };
        });
      },

      /**
       * 展开所有阶段
       */
      expandAll: () => {
        const { roadmapData } = get();
        if (roadmapData) {
          const allPhaseNumbers = roadmapData.phases.map(p => p.number);
          set({ expandedPhases: new Set(allPhaseNumbers) });
        }
      },

      /**
       * 收起所有阶段
       */
      collapseAll: () => {
        set({ expandedPhases: new Set() });
      },

      /**
       * 清除错误状态
       */
      clearError: () => set({ error: null }),
    }));
    ```

    Store 特性：
    - roadmapData: 缓存的路由数据
    - sessionState: 缓存的会话状态
    - loadingState: 加载状态（idle/loading/success/error）
    - expandedPhases: 存储展开的阶段编号（支持多阶段同时展开）
    - loadProgressData(): 并行加载 roadmap 和 state 数据（通过 gsd-tools CLI）
    - refreshData(): 重新加载数据
    - togglePhaseExpansion/expandPhase/collapsePhase: 控制单个阶段展开
    - expandAll/collapseAll: 控制所有阶段展开
